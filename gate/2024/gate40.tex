		\begin{enumerate}
			\item $\overline{D_{3}} \overline{D_{2}}$
			\item $\overline{D_{3}}D_{2} + \overline{D_{3}} \overline{D_{1}}$
			\item $D_{3}\overline{D_{2}} + \overline{D_{3}}D_{1}$
			\item $\overline{D_{3}} \overline{D_{1}}$
		\end{enumerate}
